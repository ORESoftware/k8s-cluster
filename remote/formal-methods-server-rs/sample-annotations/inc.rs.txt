// Example annotated Rust source consumable by dd-formal-methods-server.
//
// The dd-formal-methods-server treats the @-comments below as formal
// pre/postconditions and discharges them with Z3. The actual Rust code is
// inert from the analyser's perspective — only comments matter.
//
// This file lives under sample-annotations/ with a .txt extension so that
// Cargo does not try to treat it as a runnable example.

// @var x: Int
// @requires x > 0
// @ensures x + 1 > x
fn inc(x: i64) -> i64 {
    x + 1
}

// This one is a bug: the @ensures is not implied by the @requires.
// dd-formal-methods-server will report a postconditionViolation with the
// counterexample x = 0.
//
// @var x: Int
// @requires x >= 0
// @ensures x * x > 0
fn square_is_positive(x: i64) -> i64 {
    x * x
}

// Loop invariant example. @invariant is checked at entry; @variant must be
// non-negative under the invariant.
//
// @var i: Int
// @var n: Int
// @requires n >= 0
// @requires i == 0
// @invariant i <= n
// @invariant i >= 0
// @variant n - i
fn count_up_to(n: i64) -> i64 {
    let mut i: i64 = 0;
    while i < n {
        i += 1;
    }
    i
}

// Inline assertion example.
//
// @var a: Int
// @var b: Int
// @requires a >= 0
// @requires b >= 0
// @assert (a + b) >= a
fn add_nonneg(a: i64, b: i64) -> i64 {
    a + b
}
